Module Gsl.Poly

Polynomials

type poly = float array

Polynomial Evaluation

external eval : poly ‑> float ‑> float = "ml_gsl_poly_eval"

eval p x returns p.(0) +. p.(1) *. x +. p.(2) *. x**2 +. ... +. p.(n) *. x**n where n = Array.length p.
